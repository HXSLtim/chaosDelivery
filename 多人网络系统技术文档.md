## 一、网络架构概览

### 1.1 拓扑结构

```
        [主机玩家 - Server (Peer 1)]
           /        |         \
          /         |          \
    [客机A]     [客机B]     [客机C]
   (Peer 2)    (Peer 3)    (Peer 4)
```

- **非真P2P**：所有游戏逻辑以主机为权威（Authoritative Server）
- **Steam联机**：使用Steam P2P API建立连接，逻辑仍是主机-客机模式
- **支持人数**：2-4人
- **延迟容忍**：
  - < 150ms：流畅体验
  - 150-300ms：可玩，轻微延迟感
  - > 300ms：不推荐继续游戏

### 1.2 核心组件

```
NetworkManager (AutoLoad)
├── ENetMultiplayerPeer / SteamMultiplayerPeer
├── ConnectionHandler
├── StateSynchronizer
└── RPCDispatcher

GameWorld
├── MultiplayerSpawner (玩家生成)
├── MultiplayerSpawner (包裹生成)
├── EventManager
└── OrderManager

Player (场景)
├── CharacterBody2D/3D
├── MultiplayerSynchronizer
└── PlayerController

Package (场景)
├── RigidBody2D/3D
├── MultiplayerSynchronizer
└── PackageState
```

## 三、物理同步策略

### 3.1 同步频率表

| 计算类型 | 执行位置 | 同步方式 | 频率 | 优化策略 |
|----------|----------|----------|------|----------|
| 角色移动 | 客户端预测 + 主机验证 | RPC + State | 60Hz输入，20Hz同步 | 输入打包发送 |
| 包裹投掷 | 客户端发起 → 主机计算 | RPC触发 | 事件驱动 | 主机计算轨迹 |
| 包裹地面物理 | 主机 | State同步 | 10Hz | 睡眠优化 |

| 堆叠稳定性 | 主机 | State同步 | 5Hz | 仅变化时 |

### 3.2 关键优化策略

### 4.6 RPC 实现示例

```gdscript
# network_manager.gd
extends Node

# ========== 玩家 RPC ==========
@rpc("any_peer", "reliable")
func request_grab(package_id: String) -> void:
    if not multiplayer.is_server():
        return
    
    var sender_id = multiplayer.get_remote_sender_id()
    var player = get_player_by_peer_id(sender_id)
    var package = get_package_by_id(package_id)
    
    if not player or not package:
        confirm_grab.rpc_id(sender_id, package_id, false)
        return
    
    # 验证距离
    if player.global_position.distance_to(package.global_position) > MAX_GRAB_DISTANCE:
        confirm_grab.rpc_id(sender_id, package_id, false)
        return
    
    # 执行抓取
    package.grab_by(player)
    confirm_grab.rpc(package_id, true)

@rpc("authority", "reliable")
func confirm_grab(package_id: String, success: bool) -> void:
    # 客户端处理确认
    if success:
        play_grab_sound()

# ========== 输入同步 ==========
@rpc("any_peer", "unreliable")
func sync_input(input_vector: Vector2, timestamp: int) -> void:
    if not multiplayer.is_server():
        return
    
    var sender_id = multiplayer.get_remote_sender_id()
    var player = get_player_by_peer_id(sender_id)
    if player:
        player.apply_input(input_vector, timestamp)
```

## 五、防作弊设计

### 5.1 服务器验证规则

| 验证项目 | 服务器规则 | 容错范围 | 违规处理 |
|----------|------------|----------|----------|
| 移动速度 | 不得超过 4.5 m/s | ×1.2 (5.4 m/s) | 校正位置，记录日志 |
| 投掷距离 | 不得超过 7.0 m | - | 拒绝该次投掷RPC |
| 金币获取 | 验证订单完成状态 | - | 拒绝非法增加 |
| 包裹抓取 | 验证距离 < 2.0 m | +0.5m | 拒绝抓取请求 |
| 订单完成 | 验证包裹类型、目的地 | - | 拒绝非法完成 |
| 事件触发 | 仅主机可触发 | - | 忽略客户端请求 |
| 抛接时机 | 验证时间窗口 ±0.3s | ±0.05s | 拒绝非法抛接 |

### 5.2 验证代码示例

```gdscript
# 移动速度验证
func validate_movement(player: Node, new_position: Vector3, delta: float) -> bool:
    var distance = player.global_position.distance_to(new_position)
    var speed = distance / delta
    var max_speed = MAX_MOVE_SPEED * 1.2  # 容错20%
    
    if speed > max_speed:
        push_warning("Player %d speed hack detected: %.2f m/s" % [player.get_multiplayer_authority(), speed])
        return false
    return true

# 金币获取验证
func validate_coin_reward(player: Node, amount: int, reason: String) -> bool:
    match reason:
        "order_complete":
            # 验证订单确实完成
            var order_id = pending_rewards.get(player.get_multiplayer_authority())
            if not order_id or not is_order_valid(order_id):
                return false
        "perfect_catch":
            # 验证抛接确实成功
            if not validate_catch_event(player):
                return false
        _:
            return false
    return true
```

## 六、断线处理

### 6.1 断线场景处理

| 场景 | 处理策略 | 恢复机制 |
|------|----------|----------|
| 客机断线 | 角色变为AI托管或消失 | 支持重连，恢复游戏状态 |
| 主机断线 | 触发主机迁移 | 迁移后游戏继续 |
| 临时断线（< 5秒） | 角色冻结，等待重连 | 自动恢复 |
| 长期断线（> 30秒） | 踢出玩家，AI接管或移除 | 无法恢复 |

### 6.2 主机迁移流程

```gdscript
func _on_host_disconnected() -> void:
    # 1. 检测主机断线（心跳超时5秒）
    if multiplayer.is_server():
        return  # 我是主机，不需要迁移
    
    # 2. 选择新主机（peer_id最小的玩家）
    var new_host_id = get_lowest_peer_id()
    
    if new_host_id == multiplayer.get_unique_id():
        # 我成为新主机
        become_new_host()
    else:
        # 连接到新主机
        connect_to_new_host(new_host_id)

func become_new_host() -> void:
    # 3. 加载游戏状态（从最近快照）
    load_game_state(last_snapshot)
    
    # 4. 创建新的服务器
    var peer = ENetMultiplayerPeer.new()
    peer.create_server(PORT, MAX_CLIENTS)
    multiplayer.multiplayer_peer = peer
    
    # 5. 通知所有客机
    announce_new_host.rpc()
    
    # 6. 游戏继续
    resume_game()

@rpc("authority", "reliable")
func announce_new_host() -> void:
    # 客机重新连接到新主机
    reconnect_to_host()
```

        })

## 七、包裹ID生成策略

### 7.1 ID格式规范

```
格式: "<peer_id>_<sequence>_<timestamp>"
示例: "1_42_1710931200"

- peer_id: 生成者的网络ID（主机=1）
- sequence: 该peer的生成序号（自增）
- timestamp: 生成时间戳（秒级，用于唯一性）
```

### 7.2 ID生成器实现

```gdscript
# package_id_generator.gd
extends Node

var _sequence_counters: Dictionary = {}

func generate_package_id(creator_peer_id: int = 1) -> String:
    if not _sequence_counters.has(creator_peer_id):
        _sequence_counters[creator_peer_id] = 0
    
    _sequence_counters[creator_peer_id] += 1
    var sequence = _sequence_counters[creator_peer_id]
    var timestamp = Time.get_unix_time_from_system()
    
    return "%d_%d_%d" % [creator_peer_id, sequence, timestamp]

func reset_counter(peer_id: int) -> void:
    _sequence_counters[peer_id] = 0

func reset_all_counters() -> void:
    _sequence_counters.clear()
```

## 八、输入同步与预测

### 8.1 输入采样与发送

```gdscript
# player_controller.gd
extends CharacterBody3D

const INPUT_SEND_RATE: float = 0.05  # 20Hz
var _input_timer: float = 0.0
var _input_buffer: Array[Dictionary] = []

func _physics_process(delta: float) -> void:
    if is_multiplayer_authority():
        # 本地采样输入
        var input_vector = Input.get_vector("move_left", "move_right", "move_up", "move_down")
        
        # 添加到缓冲区
        _input_buffer.append({
            "vector": input_vector,
            "timestamp": Time.get_ticks_msec()
        })
        
        # 定时发送
        _input_timer += delta
        if _input_timer >= INPUT_SEND_RATE:
            _input_timer = 0.0
            send_input_batch()

func send_input_batch() -> void:
    if _input_buffer.is_empty():
        return
    
    # 发送输入批次到服务器
    NetworkManager.sync_input.rpc(_input_buffer)
    _input_buffer.clear()
```

### 8.2 服务器输入处理

```gdscript
# server_player_controller.gd
extends Node

const INPUT_BUFFER_SIZE: int = 6  # 100ms缓冲（20Hz × 0.1s = 2帧，留余量）
var _input_buffer: Array[Dictionary] = []

func apply_input_batch(inputs: Array[Dictionary]) -> void:
    for input in inputs:
        _input_buffer.append(input)
    
    # 保持缓冲区大小
    while _input_buffer.size() > INPUT_BUFFER_SIZE:
        _input_buffer.pop_front()

func _physics_process(delta: float) -> void:
    if _input_buffer.is_empty():
        return
    
    # 应用最旧的输入
    var input = _input_buffer.pop_front()
    apply_movement(input.vector, delta)
```

### 8.3 位置校正

```gdscript
# 服务器定期发送位置校正
const POSITION_SYNC_RATE: float = 0.05  # 20Hz

func _on_position_sync_timer() -> void:
    for player in players:
        var state = {
            "peer_id": player.get_multiplayer_authority(),
            "position": player.global_position,
            "velocity": player.velocity,
            "timestamp": Time.get_ticks_msec()
        }

@rpc("authority", "unreliable")
func correct_position(state: Dictionary) -> void:
    var player = get_player_by_peer_id(state.peer_id)
    if not player:
        return
    
    if player.is_multiplayer_authority():
        return  # 不校正本地玩家
    
    # 插值到服务器位置（避免瞬移）
    var target_pos = state.position
    var current_pos = player.global_position
    var error = current_pos.distance_to(target_pos)
    
    if error > POSITION_ERROR_THRESHOLD:
        # 大误差：直接校正
        player.global_position = target_pos
    else:
        # 小误差：平滑插值
        player.global_position = lerp(current_pos, target_pos, 0.3)
```

## 十、音效同步

### 10.1 音效分类

| 音效类型 | 同步方式 | 说明 |
|----------|----------|------|
| 关键音效（订单完成、抛接成功） | RPC reliable | 所有玩家必须听到 |
| 环境音效（传送带） | 本地播放 | 无需同步 |
| 交互音效（抓取、投掷） | RPC unreliable_ordered | 就近玩家听到 |
| 事件音效（警报） | RPC reliable | 事件触发时播放 |

### 10.2 音效RPC实现

```gdscript
# audio_manager.gd
extends Node

@rpc("authority", "reliable", "call_local")
func play_global_sound(sound_id: String, position: Vector3 = Vector3.ZERO) -> void:
    AudioPlayer.play(sound_id, position)

@rpc("authority", "unreliable_ordered", "call_local")
func play_positional_sound(sound_id: String, position: Vector3) -> void:
    # 仅当距离足够近时播放
    var listener_pos = get_listener_position()
    if listener_pos.distance_to(position) < MAX_SOUND_DISTANCE:
        AudioPlayer.play(sound_id, position)

# 使用示例
func on_order_completed(order: Order) -> void:
    play_global_sound.rpc("order_complete", order.position)

func on_package_thrown(package: Package) -> void:
    play_positional_sound.rpc("throw_whoosh", package.global_position)
```

## 十一、存档与进度同步

### 11.1 存档架构

```
存档数据
├── 本地进度（Steam云存档）
│   ├── 总金币
│   ├── 经验值/等级
│   ├── 解锁内容
│   └── 设置选项
└── 游戏会话状态（临时）
    ├── 当前局金币
    ├── 当前订单
    └── 玩家位置
```

### 11.2 进度同步规则

- **金币/经验**：游戏结束时由主机计算，客户端上报Steam
- **解锁内容**：基于总经验值本地计算，无需同步
- **游戏会话状态**：断线重连时从主机快照恢复

### 11.3 实现示例

```gdscript
# progress_manager.gd
extends Node

const SAVE_FILE_PATH: String = "user://save_game.dat"

func save_progress() -> void:
    var save_data = {
        "total_coins": PlayerData.total_coins,
        "experience": PlayerData.experience,
        "level": PlayerData.level,
        "unlocked_items": PlayerData.unlocked_items,
        "timestamp": Time.get_unix_time_from_system()
    }
    
    var file = FileAccess.open(SAVE_FILE_PATH, FileAccess.WRITE)
    file.store_var(save_data)
    file.close()
    
    # 上传到Steam云存档
    Steam.upload_cloud_file(SAVE_FILE_PATH)

func load_progress() -> void:
    # 从Steam云存档下载
    Steam.download_cloud_file(SAVE_FILE_PATH)
    
    if FileAccess.file_exists(SAVE_FILE_PATH):
        var file = FileAccess.open(SAVE_FILE_PATH, FileAccess.READ)
        var save_data = file.get_var()
        file.close()
        
        PlayerData.total_coins = save_data.total_coins
        PlayerData.experience = save_data.experience
        PlayerData.level = save_data.level
        PlayerData.unlocked_items = save_data.unlocked_items
```
